CREATE MEMORY TABLE SBI_EVENTS (
    ID INTEGER GENERATED BY DEFAULT AS IDENTITY(START WITH 100) NOT NULL,
    USER VARCHAR NOT NULL,
    CONSTRAINT XPKSBI_EVENTS PRIMARY KEY(ID)
);

CREATE MEMORY TABLE SBI_EVENTS_LOG (
	ID VARCHAR NOT NULL,
	USER VARCHAR NOT NULL,
	DATE TIMESTAMP DEFAULT 'now' NOT NULL,
	DESC VARCHAR NOT NULL,
	PARAMS VARCHAR NOT NULL,
	CONSTRAINT XPKSBI_EVENTS_LOG PRIMARY KEY(ID,USER, DATE)
);
